#include <stdio.h>   // Enables printf and scanf

int main() {

    int num, i, isPrime = 1;  // Variables for number, loop, and prime flag

    printf("Enter a number: ");
    scanf("%d", &num);        // Read the number

    /* Check if the number is positive, negative, or zero */
    if (num > 0)
        printf("The number is Positive.\n");
    else if (num < 0)
        printf("The number is Negative.\n");
    else
        printf("The number is Zero.\n");

    /* Check if the number is even or odd */
    if (num != 0) {
        if (num % 2 == 0)
            printf("The number is Even.\n");
        else
            printf("The number is Odd.\n");
    } else {
        printf("Zero is neither even nor odd.\n");
    }

    /* Prime number check (only for numbers greater than 1) */
    if (num <= 1) {
        printf("The number is Not Prime.\n");
    } else {
        for (i = 2; i <= num / 2; i++) { // Loop to check divisibility
            if (num % i == 0) {
                isPrime = 0;            // Not prime
                break;
            }
        }

        if (isPrime)
            printf("The number is Prime.\n");
        else
            printf("The number is Not Prime.\n");
    }

    return 0; // End of program
}
