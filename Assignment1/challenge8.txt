#include <stdio.h>   // Enables input and output functions

/* Define a structure to store student data */
struct Student {
    char name[50];   // Student name
    int roll;        // Roll number
    float marks;     // Marks obtained
    char grade;      // Calculated grade
};

int main() {

    int n, i;        // Number of students and loop variable

    printf("Enter number of students: ");
    scanf("%d", &n);                     // Read number of students

    struct Student s[n];                 // Array of structures

    /* Read data for each student */
    for (i = 0; i < n; i++) {
        printf("\nEnter details for student %d\n", i + 1);

        printf("Name: ");
        scanf("%s", s[i].name);          // Read name (no spaces)

        printf("Roll number: ");
        scanf("%d", &s[i].roll);         // Read roll number

        printf("Marks: ");
        scanf("%f", &s[i].marks);        // Read marks

        /* Calculate grade */
        if (s[i].marks >= 90)
            s[i].grade = 'A';
        else if (s[i].marks >= 75)
            s[i].grade = 'B';
        else if (s[i].marks >= 60)
            s[i].grade = 'C';
        else
            s[i].grade = 'D';
    }

    /* Display student information */
    printf("\n--- Student Details ---\n");
    for (i = 0; i < n; i++) {
        printf("\nName  : %s", s[i].name);
        printf("\nRoll  : %d", s[i].roll);
        printf("\nMarks : %.2f", s[i].marks);
        printf("\nGrade : %c\n", s[i].grade);
    }

    return 0;        // End of program
}
