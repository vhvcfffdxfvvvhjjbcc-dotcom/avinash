#include <stdio.h>   // For printf and scanf

/* Function to swap two numbers using pointers */
void swap(int *a, int *b) {
    int temp;

    temp = *a;   // Store value pointed by a
    *a = *b;     // Assign value pointed by b to a
    *b = temp;   // Assign temp to b
}

int main() {

    int x, y;
    int i;

    printf("Enter two numbers:\n");
    scanf("%d %d", &x, &y);

    /* Print values and addresses before swapping */
    printf("\nBefore Swapping:\n");
    printf("x = %d, Address of x = %p\n", x, (void*)&x);
    printf("y = %d, Address of y = %p\n", y, (void*)&y);

    /* Call swap function */
    swap(&x, &y);

    /* Print values and addresses after swapping */
    printf("\nAfter Swapping:\n");
    printf("x = %d, Address of x = %p\n", x, (void*)&x);
    printf("y = %d, Address of y = %p\n", y, (void*)&y);

    /* Array traversal using pointers */
    int arr[5] = {10, 20, 30, 40, 50};
    int *ptr = arr;   // Pointer points to first element of array

    printf("\nArray elements using pointers:\n");
    for (i = 0; i < 5; i++) {
        printf("Value = %d, Address = %p\n", *ptr, (void*)ptr);
        ptr++;   // Move pointer to next element
    }

    return 0;   // End of program
}
