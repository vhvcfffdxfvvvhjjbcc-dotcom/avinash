#include <stdio.h>    // For printf and scanf

int main() {

    char str[100], rev[100];   // Arrays for original and reversed string
    int i, length = 0;
    int vowels = 0;
    int isPalindrome = 1;

    printf("Enter a string: ");
    scanf("%s", str);          // Read string (no spaces)

    /* Find length of the string manually */
    for (i = 0; str[i] != '\0'; i++) {
        length++;
    }

    /* Reverse the string without using library functions */
    for (i = 0; i < length; i++) {
        rev[i] = str[length - i - 1];
    }
    rev[length] = '\0';        // Add null character at the end

    /* Count vowels */
    for (i = 0; i < length; i++) {
        char ch = str[i];
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
            ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U') {
            vowels++;
        }
    }

    /* Check palindrome */
    for (i = 0; i < length; i++) {
        if (str[i] != rev[i]) {
            isPalindrome = 0;
            break;
        }
    }

    /* Display results */
    printf("\nReversed string: %s\n", rev);
    printf("Number of vowels: %d\n", vowels);

    if (isPalindrome)
        printf("The string is a Palindrome.\n");
    else
        printf("The string is Not a Palindrome.\n");

    return 0;   // End of program
}
